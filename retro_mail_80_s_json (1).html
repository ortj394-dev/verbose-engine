# RETRO MAIL 1986 (Proyecto completo) ## 📁 1. server.js ```js const express = require("express"); const fs = require("fs"); const app = express(); app.use(express.json()); app.use(express.static(".")); function leerDB() { return JSON.parse(fs.readFileSync("db.json")); } function guardarDB(data) { fs.writeFileSync("db.json", JSON.stringify(data, null, 2)); } app.post("/registro", (req, res) => { const { usuario, password } = req.body; const db = leerDB(); if (db.usuarios.find(u => u.usuario === usuario)) { return res.send("Usuario ya existe"); } db.usuarios.push({ usuario, password, bandeja: [], enviados: [] }); guardarDB(db); res.send("Usuario creado"); }); app.post("/login", (req, res) => { const { usuario, password } = req.body; const db = leerDB(); const user = db.usuarios.find( u => u.usuario === usuario && u.password === password ); if (!user) return res.send("Error"); res.json(user.bandeja); }); app.post("/enviar", (req, res) => { const { de, para, mensaje } = req.body; const db = leerDB(); const receptor = db.usuarios.find(u => u.usuario === para); if (!receptor) return res.send("No existe usuario"); const correo = { de, mensaje, fecha: new Date().toLocaleString() }; receptor.bandeja.push(correo); const emisor = db.usuarios.find(u => u.usuario === de); if (emisor) emisor.enviados.push(correo); guardarDB(db); res.send("Enviado"); }); app.listen(3000, () => console.log("http://localhost:3000")); ``` --- ## 📁 2. db.json ```json { "usuarios": [] } ``` --- ## 📁 3. index.html ```html
